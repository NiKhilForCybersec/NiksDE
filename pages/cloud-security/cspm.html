Detection: CSPM Finding Aggregation

Sentinel KQL - Defender for Cloud Findings 
// Aggregate CSPM findings by severity and category
SecurityRecommendation
| where RecommendationState == "Unhealthy"
| summarize 
    FindingCount = count(),
    Resources = make_set(AssessedResourceId, 10)
    by RecommendationDisplayName, RecommendationSeverity, Category = RecommendationCategory
| order by 
    case(RecommendationSeverity == "High", 1, RecommendationSeverity == "Medium", 2, 3),
    FindingCount desc
| project 
    Severity = RecommendationSeverity,
    Category,
    Finding = RecommendationDisplayName,
    AffectedResources = FindingCount,
    SampleResources = Resources
Track Secure Score Changes 
// Monitor secure score degradation
SecureScores
| where TimeGenerated > ago(7d)
| summarize Score = avg(PercentageScore) by bin(TimeGenerated, 1d), SubscriptionId
| order by TimeGenerated asc
| serialize
| extend PreviousScore = prev(Score, 1)
| extend ScoreChange = Score - PreviousScore
| where ScoreChange < -5  // Alert on >5% drop
| project TimeGenerated, SubscriptionId, Score, PreviousScore, ScoreChange
| extend AlertSeverity = case(
    ScoreChange < -10, "High",
    ScoreChange < -5, "Medium",
    "Low"
)

Container Security Detections

Detect Privileged Container Deployment 
// Kubernetes audit logs - Privileged container
KubeAuditLogs
| where Verb in ("create", "update", "patch")
| where ObjectRef_Resource == "pods"
| extend PodSpec = parse_json(RequestObject)
| mv-expand Container = PodSpec.spec.containers
| where Container.securityContext.privileged == true
    or Container.securityContext.capabilities.add has "SYS_ADMIN"
    or PodSpec.spec.hostPID == true
    or PodSpec.spec.hostNetwork == true
| project 
    TimeGenerated,
    User = User_username,
    Namespace = ObjectRef_Namespace,
    PodName = ObjectRef_Name,
    ContainerName = tostring(Container.name),
    Image = tostring(Container.image),
    Privileged = Container.securityContext.privileged,
    HostPID = PodSpec.spec.hostPID,
    HostNetwork = PodSpec.spec.hostNetwork
| extend AlertSeverity = "High"
| extend MITRE = "T1611 - Escape to Host"
Detect Container Running as Root 
// Alert on containers without runAsNonRoot
KubeAuditLogs
| where Verb == "create"
| where ObjectRef_Resource == "pods"
| extend PodSpec = parse_json(RequestObject)
| mv-expand Container = PodSpec.spec.containers
| where isnull(Container.securityContext.runAsNonRoot) 
    or Container.securityContext.runAsNonRoot == false
| where isnull(Container.securityContext.runAsUser)
    or Container.securityContext.runAsUser == 0
| project 
    TimeGenerated,
    Namespace = ObjectRef_Namespace,
    PodName = ObjectRef_Name,
    ContainerName = tostring(Container.name),
    Image = tostring(Container.image)
| extend AlertSeverity = "Medium"

Building Detections from CSPM Findings

Alert on New Critical Findings 
// Alert when new critical CSPM findings appear
let PreviousFindings = SecurityRecommendation
| where TimeGenerated between (ago(2d) .. ago(1d))
| where RecommendationState == "Unhealthy"
| distinct AssessedResourceId, RecommendationName;
//
SecurityRecommendation
| where TimeGenerated > ago(1d)
| where RecommendationState == "Unhealthy"
| where RecommendationSeverity == "High"
| join kind=leftanti PreviousFindings on AssessedResourceId, RecommendationName
// These are NEW high-severity findings
| project 
    TimeGenerated,
    Finding = RecommendationDisplayName,
    Severity = RecommendationSeverity,
    Resource = AssessedResourceId,
    Category = RecommendationCategory,
    RemediationDescription
| extend AlertSeverity = "High"
| extend AlertType = "NewCSPMFinding"

Compliance Drift Detection

Monitor CIS Benchmark Compliance 
// Track CIS benchmark compliance over time
SecurityRecommendation
| where RecommendationDisplayName has "CIS"
| summarize 
    TotalControls = dcount(RecommendationName),
    PassingControls = dcountif(RecommendationName, RecommendationState == "Healthy"),
    FailingControls = dcountif(RecommendationName, RecommendationState == "Unhealthy")
    by bin(TimeGenerated, 1d), SubscriptionId
| extend CompliancePercent = round(100.0 * PassingControls / TotalControls, 1)
| order by TimeGenerated asc
| serialize
| extend PreviousCompliance = prev(CompliancePercent, 1)
| extend ComplianceDrift = CompliancePercent - PreviousCompliance
| where ComplianceDrift < -5  // Alert on >5% compliance drop
| project 
    TimeGenerated,
    SubscriptionId,
    CompliancePercent,
    PreviousCompliance,
    ComplianceDrift,
    PassingControls,
    FailingControls

let PublicStorageFinding = SecurityRecommendation
| where RecommendationDisplayName has "public" and RecommendationDisplayName has "storage"
| where RecommendationState == "Unhealthy"
| extend StorageAccount = extract(@"storageAccounts/([^/]+)", 1, AssessedResourceId)
| project StorageAccount, FindingTime = TimeGenerated;
//
AzureActivity
| where OperationNameValue has "Microsoft.Storage/storageAccounts"
| where ActivityStatusValue == "Success"
| extend StorageAccount = extract(@"storageAccounts/([^/]+)", 1, ResourceId)
| join kind=inner PublicStorageFinding on StorageAccount
| where TimeGenerated < FindingTime  // Activity before finding
| summarize arg_max(TimeGenerated, *) by StorageAccount  // Most recent change
| project 
    ConfigChangeTime = TimeGenerated,
    FindingTime,
    StorageAccount,
    ResponsibleUser = Caller,
    Operation = OperationNameValue

Interview Preparation

CSPM (Cloud Security Posture Management) focuses on configuration and compliance. It continuously assesses cloud resources against security best practices (CIS benchmarks, PCI DSS, etc.) and identifies misconfigurations like public S3 buckets, missing encryption, or overprivileged IAM roles. CSPM is about "is this configured securely?"

CWPP (Cloud Workload Protection Platform) focuses on runtime security of workloads. It monitors VMs, containers, and serverless functions for active threats like malware, cryptomining, or container escapes. CWPP answers "is something bad happening right now?"

Key difference: CSPM is proactive/preventive (find issues before exploitation), CWPP is detective/reactive (detect attacks in progress). Most organizations need both.

Prioritization framework:

  1. Exploitability: Is this misconfiguration internet-exposed? Can it be exploited remotely?
  2. Business impact: What data/systems are at risk? Is it production or dev?
  3. Attack path context: Does this finding enable further attacks? (e.g., public S3 with credentials)
  4. Compliance requirements: Does it violate regulatory requirements (PCI, HIPAA)?

Practical approach:

  • Start with public exposure findings (storage, databases, management ports)
  • Next address identity risks (no MFA, overprivileged accounts)
  • Then logging gaps (can't detect attacks without visibility)
  • Use attack path analysis to find findings that chain together

Integration points:

  • SIEM ingestion: Send CSPM findings and CWPP alerts to Sentinel/Splunk for correlation
  • Alert on new findings: Detect when new critical misconfigurations appear
  • Compliance drift: Alert when secure score drops significantly
  • Context enrichment: Use CSPM data to enrich other detections (e.g., "this IAM activity targeted a public bucket")

Detection engineering value:

  • CSPM provides the "what could be exploited" context
  • CWPP provides "what is being exploited" alerts
  • Combining both enables prioritized, contextual detections
  • Build playbooks that correlate CSPM findings with attack activity

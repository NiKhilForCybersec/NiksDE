Custom RFC Extraction

Python - SAP Log Extraction via RFC 
from pyrfc import Connection
import json
from datetime import datetime, timedelta

class SAPLogExtractor:
    def __init__(self, config: dict):
        self.conn = Connection(
            ashost=config['host'],
            sysnr=config['sysnr'],
            client=config['client'],
            user=config['user'],
            passwd=config['password']
        )
    
    def get_security_audit_log(self, from_date: str, to_date: str) -> list:
        """Extract Security Audit Log entries"""
        result = self.conn.call('RSAU_API_GET_LOG_DATA',
            I_FROM_DATE=from_date,
            I_TO_DATE=to_date,
            I_FROM_TIME='000000',
            I_TO_TIME='235959'
        )
        return result['ET_LOG_DATA']
    
    def get_change_documents(self, object_class: str, from_date: str) -> list:
        """Extract change documents for specific object class"""
        # Read CDHDR table
        result = self.conn.call('RFC_READ_TABLE',
            QUERY_TABLE='CDHDR',
            DELIMITER='|',
            OPTIONS=[
                {'TEXT': f"OBJECTCLAS = '{object_class}'"},
                {'TEXT': f"AND UDATE >= '{from_date}'"}
            ],
            FIELDS=[
                {'FIELDNAME': 'OBJECTCLAS'},
                {'FIELDNAME': 'OBJECTID'},
                {'FIELDNAME': 'CHANGENR'},
                {'FIELDNAME': 'USERNAME'},
                {'FIELDNAME': 'UDATE'},
                {'FIELDNAME': 'UTIME'}
            ]
        )
        return self._parse_table_result(result)
    
    def _parse_table_result(self, result: dict) -> list:
        """Parse RFC_READ_TABLE result into list of dicts"""
        fields = [f['FIELDNAME'] for f in result['FIELDS']]
        records = []
        for row in result['DATA']:
            values = row['WA'].split('|')
            records.append(dict(zip(fields, values)))
        return records

SAP User Requirements

⚠️ SIEM Integration User

Create a dedicated service account with minimum required authorizations:

  • S_RFC: RFC access to required function modules
  • S_TABU_DIS: Read access to log tables (RSAU*, CDHDR, CDPOS)
  • S_USER_GRP: Read access to user master data

Never use SAP_ALL or dialog user for SIEM integration.

Interview Preparation

Options:

  1. Native connector: Microsoft Sentinel SAP Solution, Splunk Add-on
  2. Custom extraction: RFC calls to extract from SAP tables
  3. SAP ETD: SAP's own security monitoring tool

Key data: Security Audit Log (authentication, transactions), Change Documents (master data changes), System Log (errors).

Considerations: Dedicated service user with least privilege, log volume planning, near real-time vs batch extraction.
